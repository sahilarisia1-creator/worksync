<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";

import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword, signOut, onAuthStateChanged }

  from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js";

import { getFirestore, doc, setDoc, getDoc, addDoc, updateDoc, deleteDoc,

  collection, query, orderBy, onSnapshot, serverTimestamp, where }

  from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";


// ============================================================

// 🔥 REPLACE THIS WITH YOUR FIREBASE CONFIG

// Get it from: Firebase Console → Project Settings → Your Apps

// ============================================================

const firebaseConfig = {

  apiKey:            "YOUR_API_KEY",

  authDomain:        "YOUR_PROJECT_ID.firebaseapp.com",

  projectId:         "YOUR_PROJECT_ID",

  storageBucket:     "YOUR_PROJECT_ID.appspot.com",

  messagingSenderId: "YOUR_SENDER_ID",

  appId:             "YOUR_APP_ID"

};

// ============================================================


const COLORS = ['c0','c1','c2','c3','c4','c5'];

let app, auth, db;

let currentUser = null;

let loginTime   = null;

let sessionInterval = null;

let taskFilter  = 'all';

let allTasks    = [];

let allMembers  = [];

let unsubTasks  = null;

let unsubMembers = null;


// Check if config is real

function isConfigured() {

  return firebaseConfig.apiKey && firebaseConfig.apiKey !== "YOUR_API_KEY";

}


// INIT

try {

  if (!isConfigured()) {

    document.getElementById('setupBanner').style.display = 'block';

    document.getElementById('loadingScreen').style.display = 'none';

    document.getElementById('loginView').classList.add('active');

    disableButtons();

  } else {

    app  = initializeApp(firebaseConfig);

    auth = getAuth(app);

    db   = getFirestore(app);


    onAuthStateChanged(auth, async (user) => {

      document.getElementById('loadingScreen').style.display = 'none';

      if (user) {

        const snap = await getDoc(doc(db, 'users', user.uid));

        if (snap.exists()) {

          currentUser = { uid: user.uid, ...snap.data() };

          loginTime = new Date();

          await updateDoc(doc(db, 'users', user.uid), { online: true, lastSeen: serverTimestamp() });

          showApp();

        } else {

          showLogin();

        }

      } else {

        showLogin();

      }

    });

  }

} catch(e) {

  document.getElementById('loadingScreen').style.display = 'none';

  document.getElementById('setupBanner').style.display = 'block';

  document.getElementById('loginView').classList.add('active');

  disableButtons();

  console.error(e);

}


function disableButtons() {

  document.querySelectorAll('.btn-primary,.btn-assign').forEach(b => {

    b.disabled = true;

    b.title = 'Configure Firebase first';

  });

}


// ============================================================

// AUTH

// ============================================================

window.switchLoginTab = (tab, el) => {

  document.querySelectorAll('.login-tab').forEach(t => t.classList.remove('active'));

  el.classList.add('active');

  document.getElementById('loginForm').style.display    = tab === 'login'    ? 'block' : 'none';

  document.getElementById('registerForm').style.display = tab === 'register' ? 'block' : 'none';

};


window.doLogin = async () => {

  if (!isConfigured()) { toast('Configure Firebase first!', 'error'); return; }

  const email = document.getElementById('loginEmail').value.trim();

  const pass  = document.getElementById('loginPass').value;

  if (!email || !pass) { toast('Enter email and password', 'error'); return; }

  const btn = document.getElementById('loginBtn');

  btn.disabled = true; btn.textContent = 'Signing in...';

  try {

    await signInWithEmailAndPassword(auth, email, pass);

  } catch(e) {

    toast(e.code === 'auth/invalid-credential' ? 'Wrong email or password' : e.message, 'error');

    btn.disabled = false; btn.textContent = 'Sign In →';

  }

};


window.doRegister = async () => {

  if (!isConfigured()) { toast('Configure Firebase first!', 'error'); return; }

  const name  = document.getElementById('regName').value.trim();

  const email = document.getElementById('regEmail').value.trim();

  const phone = document.getElementById('regPhone').value.trim();

  const role  = document.getElementById('regRole').value.trim();

  const pass  = document.getElementById('regPass').value;

  if (!name || !email || !pass) { toast('Fill in name, email and password', 'error'); return; }

  const btn = document.getElementById('registerBtn');

  btn.disabled = true; btn.textContent = 'Creating...';

  try {

    const cred = await createUserWithEmailAndPassword(auth, email, pass);

    const colorIdx = Math.floor(Math.random() * 6);

    await setDoc(doc(db, 'users', cred.user.uid), {

      uid: cred.user.uid, name, email, phone: phone || '',

      role: role || 'Team Member', colorIdx,

      isAdmin: false, online: false,

      lastSeen: serverTimestamp(), createdAt: serverTimestamp()

    });

    toast(`Welcome, ${name}! Signing you in...`, 'success');

  } catch(e) {

    toast(e.code === 'auth/email-already-in-use' ? 'Email already registered' : e.message, 'error');

    btn.disabled = false; btn.textContent = 'Create Account →';

  }

};


window.doLogout = async () => {

  if (!currentUser) return;

  const dur = Math.floor((new Date() - loginTime) / 1000);

  await addDoc(collection(db, 'sessions'), {

    userId: currentUser.uid, name: currentUser.name,

    loginAt: loginTime.toISOString(), duration: dur,

    date: loginTime.toLocaleDateString('en-IN'),

    createdAt: serverTimestamp()

  });

  await updateDoc(doc(db, 'users', currentUser.uid), { online: false, lastSeen: serverTimestamp() });

  clearInterval(sessionInterval);

  if (unsubTasks)   unsubTasks();

  if (unsubMembers) unsubMembers();

  currentUser = null;

  await signOut(auth);

};

function showApp() {

  document.getElementById('loginView').classList.remove('active');

  document.getElementById('appView').classList.add('active');

  initApp();

}


function showLogin() {

  document.getElementById('appView').classList.remove('active');

  document.getElementById('loginView').classList.add('active');

}


function initApp() {

  const h = new Date().getHours();

  document.getElementById('greetingTime').textContent  = h < 12 ? 'morning' : h < 17 ? 'afternoon' : 'evening';

  document.getElementById('greetingName').textContent  = `Welcome back, ${currentUser.name}`;

  document.getElementById('currentDate').textContent   =

    new Date().toLocaleDateString('en-IN', { weekday:'long', day:'numeric', month:'long', year:'numeric' });

  document.getElementById('topbarAvatar').textContent  = currentUser.name.charAt(0).toUpperCase();

  document.getElementById('topbarAvatar').className    = `avatar ${COLORS[currentUser.colorIdx % 6]}`;

  document.getElementById('topbarName').textContent    = currentUser.name.split(' ')[0];


  // Default dates

  const tomorrow = new Date(); tomorrow.setDate(tomorrow.getDate() + 1);

  const ds = tomorrow.toISOString().split('T')[0];

  ['q-date','m-date'].forEach(id => document.getElementById(id).value = ds);

  ['q-time','m-time'].forEach(id => document.getElementById(id).value = '09:00');


  startSessionTimer();

  subscribeRealtime();

}


function startSessionTimer() {

  clearInterval(sessionInterval);

  sessionInterval = setInterval(() => {

    const s = Math.floor((new Date() - loginTime) / 1000);

    const m = Math.floor(s / 60);

    document.getElementById('sessionTimer').textContent = `${m}:${(s%60).toString().padStart(2,'0')}`;

  }, 1000);

}


// ============================================================

// REAL-TIME LISTENERS

// ============================================================

function subscribeRealtime() {

  // Tasks — live updates

  const tq = query(collection(db, 'tasks'), orderBy('createdAt', 'desc'));

  unsubTasks = onSnapshot(tq, snap => {

    allTasks = snap.docs.map(d => ({ id: d.id, ...d.data() }));

    renderDashboard();

    renderAllTasks();

  });


  // Users — live presence

  unsubMembers = onSnapshot(collection(db, 'users'), snap => {

    allMembers = snap.docs.map(d => ({ uid: d.id, ...d.data() }));

    populateAssigneeDropdowns();

    renderTeam();

    renderDashboard();

  });

}


// ============================================================

// TASKS

// ============================================================

function populateAssigneeDropdowns() {

  const opts = '<option value="">— Select member —</option>' +

    allMembers.map(u =>

      `<option value="${u.uid}">${u.name}${u.uid===currentUser.uid?' (You)':''}</option>`

    ).join('');

  ['q-assignee','m-assignee'].forEach(id => {

    const el = document.getElementById(id);

    if (el) el.innerHTML = opts;

  });

}


window.assignTask = async (fromModal) => {

  const pfx = fromModal ? 'm' : 'q';

  const title      = document.getElementById(`${pfx}-title`).value.trim();

  const assigneeId = document.getElementById(`${pfx}-assignee`).value;

  const priority   = document.getElementById(`${pfx}-priority`).value;

  const date       = document.getElementById(`${pfx}-date`).value;

  const time       = document.getElementById(`${pfx}-time`).value;

  const desc       = fromModal ? document.getElementById('m-desc').value.trim() : (document.getElementById('q-note')?.value?.trim() || '');

  const channel    = fromModal ? document.getElementById('m-channel').value : 'whatsapp';


  if (!title)      { toast('Enter a task title!', 'error'); return; }

  if (!assigneeId) { toast('Select who to assign to!', 'error'); return; }


  const assignee    = allMembers.find(u => u.uid === assigneeId);

  const dueDateTime = date && time ? new Date(`${date}T${time}`).toISOString() : null;


  const btn = document.getElementById(fromModal ? 'assignBtn' : 'assignBtn') || document.querySelector('.btn-assign');


  try {

    await addDoc(collection(db, 'tasks'), {

      title, desc,

      assigneeId, assigneeName: assignee.name,

      assignedById: currentUser.uid, assignedByName: currentUser.name,

      priority, channel, dueDateTime,

      status: 'pending',

      createdAt: serverTimestamp()

    });


    // Log alert

    await addDoc(collection(db, 'alerts'), {

      type: 'Task Assigned', from: currentUser.name, to: assignee.name,

      task: title, channel, time: serverTimestamp()

    });


    // WhatsApp

    const msg = `📌 *New Task from ${currentUser.name}*\n\n*Task:* ${title}${desc?`\n*Details:* ${desc}`:''}\n*Priority:* ${priority.toUpperCase()}\n*Due:* ${dueDateTime ? new Date(dueDateTime).toLocaleString('en-IN') : 'Flexible'}\n\nPlease confirm receipt. — WorkSync`;

    if (assignee.phone) openWA(assignee.phone, msg);


    toast(`Task assigned to ${assignee.name}!`, 'success');


    document.getElementById(`${pfx}-title`).value = '';

    if (fromModal) { document.getElementById('m-desc').value = ''; closeAssignModal(); }

    else if (document.getElementById('q-note')) document.getElementById('q-note').value = '';

  } catch(e) {

    toast('Error saving task: ' + e.message, 'error');

  }

};


window.markDone = async (id) => {

  const t = allTasks.find(t => t.id === id);

  if (!t) return;

  try {

    await updateDoc(doc(db, 'tasks', id), { status: 'done', doneAt: serverTimestamp() });

    await addDoc(collection(db, 'alerts'), {

      type: 'Task Done', from: currentUser.name, to: t.assignedByName,

      task: t.title, channel: 'system', time: serverTimestamp()

    });

    const assigner = allMembers.find(u => u.uid === t.assignedById);

    if (assigner && assigner.phone) {

      const msg = `✅ *Task Completed!*\n\n${currentUser.name} completed: *${t.title}*\n\n— WorkSync`;

      openWA(assigner.phone, msg);

    }

    toast('Task marked as done!', 'success');

  } catch(e) {

    toast('Error: ' + e.message, 'error');

  }

};


window.sendReminder = async (id) => {

  const t = allTasks.find(t => t.id === id);

  if (!t) return;

  const assignee = allMembers.find(u => u.uid === t.assigneeId);

  if (!assignee) { toast('Member not found', 'error'); return; }

  const msg = `🔔 *Reminder from ${currentUser.name}*\n\nPlease update on: *${t.title}*\nPriority: ${t.priority.toUpperCase()}${t.dueDateTime?'\nDue: '+new Date(t.dueDateTime).toLocaleString('en-IN'):''}\n\n— WorkSync`;

  if (assignee.phone) openWA(assignee.phone, msg);

  await addDoc(collection(db, 'alerts'), {

    type: 'Reminder Sent', from: currentUser.name, to: assignee.name,

    task: t.title, channel: 'whatsapp', time: serverTimestamp()

  });

  toast(`Reminder sent to ${assignee.name} via WhatsApp!`, 'wa');

};


window.deleteTask = async (id) => {

  if (!confirm('Delete this task?')) return;

  await deleteDoc(doc(db, 'tasks', id));

};

  el.classList.add('active');

  renderAllTasks();

};

function renderAllTasks() {

  let filtered = allTasks;

  if (taskFilter==='mine')    filtered = allTasks.filter(t => t.assigneeId===currentUser.uid);

  else if (taskFilter==='urgent')  filtered = allTasks.filter(t => t.priority==='urgent' && t.status!=='done');

  else if (taskFilter==='pending') filtered = allTasks.filter(t => t.status==='pending');

  else if (taskFilter==='done')    filtered = allTasks.filter(t => t.status==='done');


  document.getElementById('allTasksList').innerHTML = filtered.length

    ? filtered.map(t => taskCardHTML(t)).join('')

    : `<div class="empty-state"><span class="empty-icon">📭</span><div class="empty-title">No tasks here</div></div>`;

}


function taskCardHTML(t) {

  const isOverdue = t.dueDateTime && new Date(t.dueDateTime) < new Date() && t.status!=='done';

  const badgeClass = t.status==='done' ? 'badge-done' : isOverdue ? 'badge-overdue' : `badge-${t.priority}`;

  const badgeLabel = t.status==='done' ? '✓ Done' : isOverdue ? '⚠ Overdue' : t.priority;

  const due = t.dueDateTime ? new Date(t.dueDateTime).toLocaleDateString('en-IN',{day:'numeric',month:'short',hour:'2-digit',minute:'2-digit'}) : null;

  const isMine   = t.assigneeId === currentUser.uid;

  const isAuthor = t.assignedById === currentUser.uid;

  const canAct   = isMine || isAuthor || currentUser.isAdmin;


  return `<div class="task-card ${t.status==='done'?'done':t.priority}">

    <div class="task-top">

      <div class="task-title-text">${t.title}</div>

      <div class="badges"><span class="badge ${badgeClass}">${badgeLabel}</span></div>

    </div>

    ${t.desc?`<div style="font-size:0.77rem;color:var(--muted);margin-bottom:8px;line-height:1.4;">${t.desc}</div>`:''}

    <div class="task-meta">

      <span>👤 ${t.assigneeName}</span>

      <span>📤 ${t.assignedByName}</span>

      ${due?`<span>🗓 ${due}</span>`:''}

    </div>

    ${canAct?`<div class="task-actions">

      ${t.status!=='done' && isMine ? `<button class="action-btn green" onclick="markDone('${t.id}')">✓ Done</button>`:''}

      ${t.status!=='done' ? `<button class="action-btn wa" onclick="sendReminder('${t.id}')">💬 Remind</button>`:''}

      ${isAuthor||currentUser.isAdmin ? `<button class="action-btn red" onclick="deleteTask('${t.id}')">✕</button>`:''}

    </div>`:''}

  </div>`;

}

window.renderActivity = async () => {

  // Sessions

  const sSnap = await import("https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js").then(m =>

    m.getDocs(m.query(collection(db,'sessions'), orderBy('createdAt','desc'), m.limit(20)))

  );

  const rows = sSnap.docs.map(d => d.data());

  document.getElementById('sessionTableBody').innerHTML = rows.length

    ? rows.map(s => {

        const m2 = Math.floor(s.duration/60), s2 = s.duration%60;

        const memberColor = allMembers.find(u=>u.uid===s.userId);

        return `<tr>

          <td><div style="display:flex;align-items:center;gap:8px;">

            <div class="avatar ${memberColor?COLORS[memberColor.colorIdx%6]:'c0'}" style="width:28px;height:28px;border-radius:7px;font-size:0.7rem;">${s.name.charAt(0)}</div>

            ${s.name}</div></td>

          <td>${s.loginAt ? new Date(s.loginAt).toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}) : '—'}</td>

          <td style="color:var(--accent2)">${m2}m ${s2}s</td>

          <td style="color:var(--muted)">${s.date||'—'}</td>

        </tr>`;

      }).join('')

    : '<tr><td colspan="4" style="text-align:center;color:var(--muted);padding:24px;">No sessions yet</td></tr>';


  // Alerts

  const { getDocs, query: q2, limit } = await import("https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js");

  const aSnap = await getDocs(q2(collection(db,'alerts'), orderBy('time','desc'), limit(30)));

  const aList = aSnap.docs.map(d => d.data());

  const icons = {'Task Assigned':'📋','Task Done':'✅','Reminder Sent':'🔔'};

  document.getElementById('activityList').innerHTML = aList.length

    ? aList.map(a => `<div class="activity-item">

        <div class="activity-icon" style="background:rgba(59,130,246,0.1);">${icons[a.type]||'📢'}</div>

        <div style="flex:1;font-size:0.8rem;line-height:1.4;"><strong style="color:var(--accent2)">${a.from}</strong> → <strong style="color:var(--accent2)">${a.to}</strong>: ${a.type} — "${a.task}"</div>

        <div class="activity-time">${a.time?.toDate ? a.time.toDate().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}) : '—'}</div>

      </div>`).join('')

    : `<div class="empty-state"><span class="empty-icon">📭</span><div class="empty-title">No activity yet</div></div>`;

};

window.switchTab = (tab, el) => {

  document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));

  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));

  document.querySelectorAll(`.nav-tab`).forEach(t => {

    if (t.getAttribute('onclick')?.includes(`'${tab}'`)) t.classList.add('active');

  });

  document.getElementById(`tab-${tab}`).classList.add('active');

  if (tab === 'activity') window.renderActivity();

};

  const icons  = {success:'✅',error:'❌',wa:'💬',default:'📌'};

  el.innerHTML = `<div class="toast-title">${icons[type]} ${titles[type]}</div>${msg}`;

  stack.appendChild(el);

};
